Operator overloading in C++

Binaire operatoren

/* binaire operatoren overladen*/

class ComPlex
{
	double re,im;
public:
	ComPlex(){
		re=0;
		im=0;
	}
	ComPlex(double r, double i){
		re=r;
		im=i;
	}
	ComPlex operator+(ComPlex); 	//operator definitie 
									//compiler vertaling: a.operator+(b)
									//member methode

	friend ComPlex operator*(ComPlex,ComPlex); 	//friend operator definitie
												//compiler vertaling: operator*(a,b)
												//losstaande methode
};
/* ************* einde klasse ***************** */

/* returnType klasseNaam:: methode */
ComPlex ComPlex:: operator+(ComPlex ander){
	return ComPlex(re+ander.re , im+ander.im); //return nieuw complex getal
}
ComPlex ComPlex:: operator*(ComPlex a, ComPlex b){
	/* ERROR: overloaded 'operator*' must be a unary or binary operator (has 3 parameters) */
	/* iemand een oplossing? */

	ComPlex res;
	res.re= (a.re*b.re) - (a.im*b.im);
	res.im= (a.im*b.re) - (a.re*b.im);
	return res;
}

int main(){
	ComPlex a(4,1);
	ComPlex b(2,1);
	ComPlex c,d;

	c=a+b;
	d=a*b;

	return 0;
}

#include <iostream>
using namespace std;
	class RangeInt{
		int waarde;
		int onder,boven;
		public:
		RangeInt(int o, int b, int w){
			onder =o;
			boven =b;
			waarde=w;
		}
		void schrijf(){
			cout << waarde <<endl; }

		/* member */
		RangeInt& operator++(); 	//prefix
		//vertalign compiler: a.operator@()
		//?? voor wat staat daar een & ??
		RangeInt operator++(int dummy); //postfix
		//vertalign compiler: a.operator@(i)
		//?? kunnen we niet gebruik maken van ++operator() ??
		/* niet-member */
		friend RangeInt& operator--(RangeInt& ri); 	//prefix
		//vertalign compiler: operator@(a)
		friend RangeInt operator--(RangeInt ri, int Dummy); //postfix
		//vertalign compiler: operator@(a,i)
	};
	RangeInt& RangeInt::operator++(){
		if(waarde<boven) waarde++;
		else{
		cout<< "te groot\n";	//foutafhandeling
		}
		return *this;	//return by reference
	//http://stackoverflow.com/questions/6919330/return-this-in-c
	}
	RangeInt RangeInt::operator++(int Dummy){
		//eerst huidige waarde opslaan
		//dan object wijzigen
		//opgeslagen waarde teruggeven
		RangeInt tmp(*this);	//default copyconstructor
		//huidige waarde opslaan dus

		if(waarde<boven) waarde ++; //object wijzigen
		else{
		cout <<"te groot\n";
		}
		return tmp; //RangeInt met oude waarde terug afgeven
	}
	RangeInt& operator--(RangeInt& ri){
		if(ri.waarde> ri.onder) ri.waarde --;
		else{
		cout <<"te klein\n";
		}
		return ri;
	}
	RangeInt operator--(RangeInt ri, int Dummy){
		RangeInt tmp(ri); //default copyconstructor
		if(ri.waarde> ri.onder) ri.waarde --;
		else{
		cout <<"te klein\n";
		}
		return tmp;	//return tmp van het type RangeInt
	}


/******************** main *************************/
int main(){
// onder, boven, waarde
RangeInt a(5,10,7), b(5,10,5);
a.schrijf();				// 7
b.schrijf();				// 5
(++a).schrijf();			// 8
b=a++;	
a.schrijf();				// 9
b.schrijf();				// 8
(--a).schrijf();			// 8
b=a--;
a.schrijf();				// 8
b.schrijf();				// 8
return 0;
}

Vragen:

[1] Waarom maken we gebruik van het adresseringssymbool '&', bij RangeInt& operator++(); ?


Antwoord:
Vermits we willen verder werken met het complex getal moeten we de referentie terug krijgen en niet de gekopieerde waarde. Zodat het complex getal een lvalue is. Deze kan links staan van de '=' operator.
[2] Kunnen we bij C++ niet gebruik maken van ++operator() om duidelijk te maken dat het om een postfix gaat?


Antwoord:
Neen, blijkbaar niet.
[3] Wat verduidelijking bij het return type in de RangeInt& RangeInt::operator++() methode.


Antwoord:
Zie vraag [1], men moet een RangeInt& (call by reference) hebben als return type vermits er geen argumenten worden meegegeven.
[4] Returnen we het adres, ri in de RangeInt& operator--(RangeInt& ri) methode?


Antwoord:
Neen, zoals de vragen hierboven hebben we te maken met een call-by-reference zodat we het kunnen gebruiken als lvalue, zie vraag [1]. Met andere woorden we kunnen meerdere malen incrementeren omdat het gaat over hetzelfde object. (++(++(++complex_getalltje)))
